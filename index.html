// SUBMIT RSVP

async function submitRSVP() {

    if (!currentGuestRecordId) {

        alert("Please search for your name first.");

        return;

    }


    // VALIDATION: Check if dance song field is required but empty

    const danceYesBtn = document.querySelector('input[name="dance_pref"][value="Yes"]');

    const danceSongInput = document.getElementById('dj_song_dance');

    

    if (danceYesBtn && danceYesBtn.checked && !danceSongInput.value.trim()) {

        alert("Please provide a song name if you're planning to dance!");

        danceSongInput.focus();

        return;

    }


    const btn = document.getElementById('finalSubmitBtn');

    btn.disabled = true;

    btn.innerText = "Processing...";


    try {

        // Collect all event selections

        const fields = {};


        // Add event selections from the event RSVPs list

        ['cocktail', 'haldi', 'pelli', 'muhurtham'].forEach(event => {

            const rsvpSelect = document.getElementById(`${event}_rsvp`);

            const countSelect = document.getElementById(`${event}_count`);

            if (rsvpSelect && countSelect) {

                fields[`${event.charAt(0).toUpperCase() + event.slice(1)}_RSVP`] = rsvpSelect.value;

                fields[`${event.charAt(0).toUpperCase() + event.slice(1)}_Count`] = parseInt(countSelect.value);

            }

        });


        // Add dance preferences (for everyone)

        const dancePref = document.querySelector('input[name="dance_pref"]:checked');

        if (dancePref) fields["Cocktail_Dance"] = dancePref.value;

        

        // Add general DJ song request (optional)

        const djSongGeneralInput = document.getElementById('dj_song_general');

        if (djSongGeneralInput && djSongGeneralInput.value.trim()) fields["Cocktail_DJ_Song"] = djSongGeneralInput.value.trim();


        // Add dance song request (mandatory if dancing)

        const djSongDanceInput = document.getElementById('dj_song_dance');

        if (djSongDanceInput && djSongDanceInput.value.trim()) fields["Cocktail_Dance_Song"] = djSongDanceInput.value.trim();


        // Add preferences section data if visible (Cocktail specific)

        if (!document.getElementById('preferences-section').classList.contains('hidden')) {

            const cocktailCount = parseInt(document.getElementById('cocktail_count')?.value || 0);

            

            // Bar Preference: Map to count fields

            const barPref = document.getElementById('alcohol_choice').value;

            if (barPref === "Alcohol") {

                fields["Cocktail_Alcohol_Count"] = cocktailCount;

                fields["Cocktail_Mocktail_Count"] = 0;

            } else {

                fields["Cocktail_Mocktail_Count"] = cocktailCount;

                fields["Cocktail_Alcohol_Count"] = 0;

            }

            

            // Meal Preference: Map to count fields

            const mealPref = document.querySelector('input[name="meal_pref"]:checked');

            if (mealPref) {

                if (mealPref.value === "Veg") {

                    fields["Cocktail_Veg_Count"] = cocktailCount;

                    fields["Cocktail_NonVeg_Count"] = 0;

                } else {

                    fields["Cocktail_NonVeg_Count"] = cocktailCount;

                    fields["Cocktail_Veg_Count"] = 0;

                }

            }

        }


        console.log("Sending to Airtable:", JSON.stringify({ fields }, null, 2));

        console.log("Record ID:", currentGuestRecordId);


        const response = await fetch(`https://api.airtable.com/v0/${BASE_ID}/${TABLE_NAME}/${currentGuestRecordId}`, {

            method: 'PATCH',

            headers: {

                'Authorization': `Bearer ${API_TOKEN}`,

                'Content-Type': 'application/json'

            },

            body: JSON.stringify({ fields })

        });


        const responseData = await response.json();

        console.log("Airtable Response:", responseData);


        if (response.ok) {

            showRSVPModal();

            document.getElementById('event-display-section').classList.add('hidden');

        } else {

            console.error("Airtable Error Status:", response.status);

            console.error("Airtable Error Details:", responseData);

            throw new Error(responseData.error?.message || "Update failed");

        }

    } catch (e) {

        console.error("Full Error:", e.message);

        alert("Error: " + e.message + "\n\nPlease check the console (F12) for details and contact support.");

        btn.disabled = false;

        btn.innerText = "Confirm All Selections";

    }

}


// MODAL FUNCTIONS

function showRSVPModal() {

    const modal = document.getElementById('rsvpModal');

    const card = document.getElementById('modalCard');

    

    modal.classList.remove('hidden');

    setTimeout(() => {

        card.classList.remove('scale-95', 'opacity-0');

        card.classList.add('scale-100', 'opacity-100');

    }, 10);

}


function closeRSVPModal() {

    const modal = document.getElementById('rsvpModal');

    const card = document.getElementById('modalCard');

    

    card.classList.add('scale-95', 'opacity-0');

    setTimeout(() => {

        modal.classList.add('hidden');

        document.getElementById('event-display-section').classList.add('hidden');

        document.getElementById('guestName').value = '';

        window.scrollTo({ top: 0, behavior: 'smooth' });

    }, 300);

}


// PHOTO UPLOADER

function toggleUploader() {

    const container = document.getElementById('uploaderContainer');

    

    if (container.classList.contains('hidden')) {

        container.classList.remove('hidden');

        container.scrollIntoView({ behavior: 'smooth', block: 'start' });

    } else {

        container.classList.add('hidden');

    }

}


// TOGGLE DANCE SONG REQUEST FIELD BASED ON DANCE PREFERENCE

function toggleDanceSongField() {

    const danceYesBtn = document.querySelector('input[name="dance_pref"][value="Yes"]');

    const danceSongField = document.getElementById('dance-song-request-field');

    const danceSongInput = document.getElementById('dj_song_dance');

    

    if (!danceYesBtn) return; // Exit if button doesn't exist

    

    if (danceYesBtn.checked) {

        danceSongField.classList.remove('hidden');

        danceSongInput.focus(); // Auto-focus on the song input field

    } else {

        danceSongField.classList.add('hidden');

        danceSongInput.value = ''; // Clear the field when not dancing

    }

}
